오늘 배웠던 내용에 대해 MD파일로 작성해보는 시간
# 최초 실행
git config --global user.name "github username"
git config --global user.email "github email"

# 일반 폴더 -> 로컬 저장소
git init

# 버전 상태 출력
git status

# Working Directory -> Staging Area
git add.

# staging Area -> Commits
git commit -m "first commit"

# comits 목록 출력
git log
git log --online
git log -p
